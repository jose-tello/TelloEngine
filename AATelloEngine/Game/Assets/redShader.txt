__VERTEX_SHADER__ 
#version 330 core
layout (location = 0) in vec3 position;

uniform mat4 model_matrix;
uniform mat4 view;
uniform mat4 projection;

void main()
{
 gl_Position = projection * view * model_matrix * vec4(position, 1.0f);
}


__FRAGMENT_SHADER__
#version 330 core
out vec4 color;

void main()
{
  color = vec4(1.0, 0.0, 0.0, 1.0);
}
